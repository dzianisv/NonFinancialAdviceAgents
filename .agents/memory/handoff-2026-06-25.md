```python
# Run to check:
/Users/engineer/.venv/bin/python3 << 'EOF'
import yfinance as yf, datetime, warnings; warnings.filterwarnings('ignore')
results = {}
for sym in ['ROBO','IRBO']:
    h = yf.Ticker(sym).history(period='90d',interval='1d',auto_adjust=True)
    price=float(h['Close'].iloc[-1]); ma50=float(h['Close'].rolling(50).mean().iloc[-1])
    d=h['Close'].diff(); g=d.clip(lower=0).ewm(com=13,adjust=False).mean()
    l=(-d.clip(upper=0)).ewm(com=13,adjust=False).mean()
    rsi=float(100-100/(1+g.iloc[-1]/l.iloc[-1]))
    e12=h['Close'].ewm(span=12,adjust=False).mean(); e26=h['Close'].ewm(span=26,adjust=False).mean()
    macd=e12-e26; sig=macd.ewm(span=9,adjust=False).mean()
    hist=float((macd-sig).iloc[-1]); prev=float((macd-sig).iloc[-2])
    results[sym]=dict(price=price,ma50=ma50,rsi=rsi,hist=hist,prev=prev)
print('ROBO/IRBO WATCH',datetime.date.today())
r=results['ROBO']; i=results['IRBO']
for sym,d,trig,abort in [('ROBO',r,r['price']>86 and r['hist']>0,r['price']<76.5),
                          ('IRBO',i,i['hist']>0,i['price']<66)]:
    s='ABORT' if abort else ('TRIGGER-ENTER' if trig else ('WATCH' if d['hist']>d['prev'] else 'WAIT'))
    print(f"{sym} ${d['price']:.2f} RSI {d['rsi']:.1f} MACDhist {d['hist']:+.3f} => {s}")
EOF
```
